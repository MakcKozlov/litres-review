```dataviewjs
const pages = dv.pages('"Books"').where(p => p.type === "book-review").sort(p => p.date, "desc");

const container = dv.el("div", "", {attr: {style: "display:grid;grid-template-columns:repeat(auto-fill,160px);gap:16px;padding:8px 0"}});

for (const p of pages) {
  const coverPath = app.vault.adapter.getResourcePath(p.file.folder + "/cover.jpg");
  
  const card = container.createEl("div", {attr: {
    style: "width:160px;height:240px;border-radius:10px;overflow:hidden;box-shadow:0 2px 8px rgba(0,0,0,0.3);cursor:pointer"
  }});
  
  card.createEl("img", {attr: {
    src: coverPath,
    style: "width:100%;height:100%;object-fit:cover"
  }});
  
  card.addEventListener("click", () => {
    app.workspace.openLinkText(p.file.path, "");
  });
}
```
